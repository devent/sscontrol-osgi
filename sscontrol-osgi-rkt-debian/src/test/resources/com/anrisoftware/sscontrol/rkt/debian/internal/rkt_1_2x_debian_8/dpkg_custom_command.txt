#!/bin/bash
theCommand="$0"
echo "$theCommand $@"
echo "$@" >> $theCommand.out
echo "$1"
if [[ "$1" == "-i" ]]; then
    echo "Return 0"
    exit 0
fi
if [[ "$1" == "-s" ]]; then
    echo "Return 1"
    exit 1
fi
if [ "$#" -gt 0 ]; then
    cmd=($1)
    cmd=${cmd[0]}
    if command -v -- "$cmd"&>/dev/null; then
        $@
    fi
fi
