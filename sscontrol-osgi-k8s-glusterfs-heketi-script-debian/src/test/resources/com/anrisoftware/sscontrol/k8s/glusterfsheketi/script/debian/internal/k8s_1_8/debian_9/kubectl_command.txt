#!/bin/bash
theCommand="$0"
echo "$theCommand $@"
echo "$@" >> $theCommand.out
if [ "$#" -gt 0 ]; then
    cmd=($1)
    cmd=${cmd[0]}
    if command -v -- "$cmd"&>/dev/null; then
        if [[ "$cmd" != "enable" ]]; then
            $@
        fi
    fi
fi

echo $PWD
if [ ! -f "_kubectl" ]; then
    echo "a" > "_kubectl"
    echo 'error: no context exists with the name: "xx".'
    exit 1
fi
